# Base command for work
git init - инициализация проекта.
git add . - добавления всех файлов/а в коммит.
git commit '' - фиксация изменений.
git push - закинуть коммиты на удалённый репозиторий.
git pull - забрать изменения из репозитория.

# Other
git clone - клонировать репозиторий ( можно пропустить инициализацию )
git branch - узнать какие ветки существуют в репозитории.
git checkout - помогает для перемещения между ветками. (switch обновлённая версия)
git merge - слить две ветки. (надо находиться через checkout или switch на одной ветки, а в merge писать другую)
git rebase - слияние веток. (но делает это чище по сравнению с merge) Больше для личных веток не для совместной работой над одной.
git fetch - просмотреть изменения, не изменяя свои файлы.
git diff - посмотреть различия в ветке.
git revert <hash> - изменит commit перевернув его, но сохранит целостность.

# Information
git log --oneline --graph - посмотреть визуализацию веток. 
git status - посмотреть состояние ветки/файлов.
